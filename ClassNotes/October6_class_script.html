library(ggplot2)

gdURL <- "http://tiny.cc/gapminder"
data <- read.delim(file = gdURL) 

j_coefs <- ddply(data, ~country + continent, 
                 function(x, offset = 1952){
                   the_fit  <- lm(lifeExp ~ I(year-offset), x)
                   setNames(coef(the_fit), c("intercept", "slope"))
                 })
